Base URL:

https://us-central1-picpr-z.cloudfunctions.net/webApi/api/v1
# Lien Methode Description Params
# /users POST Permet d'ajouter les utilisateurs authentifiés

age : Entier

displayName : String

email : String

phoneNumber : String

online : long

photoURL : String

uid : String
# /users GET Récupérer la liste des utilisateurs authentifiés
# /contact POST Permet d'ajouter un contact

senderId : String (id de celui qui envoie la requête)

receiverId : String (id de celui à qui la demande est destinée)

# /contacts/{contactId} DELETE Permet de supprimer un contact

contactId : String (id de la rêquête d'ajout de contact; doit rester dans le lien)

# /contacts/{contactId} PATCH Permet de bloquer un contact

contactId : String (id de la rêquête d'ajout de contact; doit rester dans le lien)

# /contacts/{userId} GET Récupérer de récupérer la liste des amis d'un utilisateur

userId : String (id de l'utilisateur; doit rester dans le lien)

# /pp-requests POST Permet d'envoyer une requête de photo

senderId : String

reveiverId : String
# /pp-feedback POST Permet d'envoyer une requête de photo

pp_requestId : String (Id de la requête)

picUrl : String (le lien de l'image uploadé, doit commencer par https)

label : String

type : String ("accepted" ou "rejected") précise si la requête est acceptée ou pas

# /pp-requests-lists POST Récupérer une conversation

senderId : String (id de celui qui envoie la requête)

receiverId : String (id de celui à qui la demande est destinée)

# /mycontacts POST Récupérer une conversation
Retourne les ids de tous les utilisateurs PicPriz de cette liste de contacts

userContacts : String (la liste des contacts séparés par ';')
